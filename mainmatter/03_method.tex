\section{アルゴリズム}
\label{sec:algorithm}

アルゴリズムの例をアルゴリズム\ref{alg:selection}に示す．

\begin{algorithm}[H]
    \caption{最大値の探索}
    \label{alg:selection}
    \begin{algorithmic}[1]
        \Require 配列 $A$，要素数 $n$
        \Ensure 最大値 $\mathrm{max}$
        \State $\mathrm{max} \gets A[1]$
        \For{$i \gets 2$ \textbf{to} $n$}
            \If{$A[i] > \mathrm{max}$}
                \State $\mathrm{max} \gets A[i]$
            \EndIf
        \EndFor
        \State \Return $\mathrm{max}$
    \end{algorithmic}
\end{algorithm}

\section{定義と定理}
\label{sec:theorem}

\begin{definition}[グラフ]
頂点集合 $V$ と辺集合 $E$ の組 $G = (V, E)$ をグラフと呼ぶ．
\label{def:graph}
\end{definition}

定義\ref{def:graph}に基づき，次の定理を示す．

\begin{theorem}
空でない有限集合 $V$ に対し，完全グラフでは任意の異なる 2 頂点間に辺が存在する．
\label{thm:complete_graph}
\end{theorem}

\begin{proof}
完全グラフの定義より，任意の異なる 2 頂点 $u, v \in V$ に対して辺 $(u, v)$ が存在する．
したがって，任意の異なる 2 頂点間に辺が存在する．
\end{proof}
